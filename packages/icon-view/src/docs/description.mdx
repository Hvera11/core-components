## SuperEllipse

### С заливкой

```tsx live
// import { Typography } from '@alfalab/core-components/typography';

const SIZES = ['48', '64', '80', '128'];

const iconViewContent = {
    '48': <Typography.Text view="primary-small" weight="bold">ЙО</Typography.Text>,
    '64': <Typography.Text view="primary-large" weight="bold">ЙО</Typography.Text>
    '80': <Typography.Title view="small" weight="bold">ЙО</Typography.Title>
    '128': <Typography.Title view="medium" weight="bold">ЙО</Typography.Title>
}

render(
    <Container>
        <Row>
            {SIZES.map(size => (
                <Col key={size}>
                    <IconView size={size}>
                        {iconViewContent[size]}
                    </IconView>
                </Col>
            ))}
        </Row>

        <Row>
            {SIZES.map(size => (
                <Col key={size}>
                    <IconView size={size} topAddons={<CheckmarkCircleMIcon color="#2fc26e" />}>
                        <SfTouchIdXxlIcon />
                    </IconView>
                </Col>
            ))}
        </Row>

        <Row>
            {SIZES.map(size => (
                <Col key={size}>
                    <IconView size={size} bottomAddons={<CheckmarkCircleMIcon color="#2fc26e" />}>
                        <ToolsMIcon />
                    </IconView>
                </Col>
            ))}
        </Row>

        <Row>
            {SIZES.map(size => (
                <Col key={size}>
                    <IconView
                        size={size}
                        topAddons={<AlertCircleMIcon color="#2fc26e" />}
                        bottomAddons={<CheckmarkCircleMIcon color="#2fc26e" />}
                    >
                        <SfPhoneXxlIcon />
                    </IconView>
                </Col>
            ))}
        </Row>
    </Container>,
);
```
